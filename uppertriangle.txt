class diag
{
	int *d;
	int n;
public:
	 diag(int size=10)
	 {
	 	 n=size;
	 	 if(size!=0)
	 	 {
	 	 	d=new int[n*(n+1)/2];
	 	 }
	 	 else
	 	 	d=0;
	 }
	 ~diag()
	 {
	 	delete []d;
	 }
	 void store(int a,int i,int j);
	 int retrieve(int i,int j);
	 void display();
	
	 
   };

void diag::store(int a,int i,int j)
{
	int k=0;
	if(i<1||i>n||j<1||j>n)
	{
		std::cout<<"\nOut of bound";
		exit(0);
	}
	else if(i>j&&a!=0)
		std::cout<<"\nMust be zero";
	else if(j>=i)
	{
        k= (j*(j-1)/2+i-1);
		d[k]=a;
	}
}
